## Instructions

<RecipeSteps
  steps={[
    text("Preheat the oven to 180°C"),
    text("Grease a baking dish with butter"),
    text(
      "Wash the rhubarb, cut into small pieces, and spread evenly in the dish",
    ),
    text("Separate the eggs"),
    text("Beat the yolks with sugar until pale and fluffy"),
    text("Whip the egg whites until stiff peaks form"),
    text("Add the flour to the yolk mixture and mix"),
    text("Fold in the whipped egg whites"),
    text("Pour the batter over the rhubarb"),
    text("Bake for 45-50 minutes until golden brown"),
  ]}
