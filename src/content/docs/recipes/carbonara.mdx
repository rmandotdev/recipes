## Instructions

<RecipeSteps
  steps={[
    text("Bring salted water to a boil"),
    text("Cook the spaghetti until al dente"),
    text(
      "While the pasta cooks, fry the bacon in sunflower oil until rendered but still soft",
    ),
    text("Separate 3 egg yolks. Add 1 whole egg"),
    text("Whisk the eggs with parmesan, black pepper and paprika"),
    text("Reserve 300ml of pasta water. Drain the spaghetti"),
    text(
      "Reduce heat to low. Add the spaghetti to the pan with the bacon. Toss",
    ),
    text("Remove from heat"),
    text("Pour the egg mixture over the pasta. Toss quickly"),
    text("Return to low heat. Stir constantly for 1 minute until creamy"),
    text("Add pasta water gradually as needed"),
    text("Serve with extra parmesan and black pepper"),
  ]}
/>
